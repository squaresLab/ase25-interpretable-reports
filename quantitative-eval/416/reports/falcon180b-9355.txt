what: The code contains a race condition due to improper handling of shared resources, specifically the `client_sockets` array and `num_clients` counter, which are accessed and modified by multiple threads without any synchronization mechanism.
where: The vulnerability is located in the `handle_client` function and the main event loop:

- main: Lines 45-59
why: If not resolved, this race condition can lead to incorrect behavior such as corrupting the `client_sockets` array, causing crashes, or allowing an attacker to manipulate connections, potentially leading to a denial of service or unauthorized access.
how: An attacker can exploit this race condition by orchestrating multiple simultaneous connections to the server, increasing the likelihood of concurrent access to `client_sockets` and `num_clients` without synchronization, which may lead to undefined behavior or crashes.

#include <netinet/in.h>
#include <pthread.h>

#define MAX_CLIENTS 10
#define BUFFER_SIZE 1024

int server_socket;
int client_sockets[MAX_CLIENTS];
int num_clients = 0;

pthread_t threads[MAX_CLIENTS];
pthread_mutex_t client_mutex = PTHREAD_MUTEX_INITIALIZER;

void *handle_client(void *arg) {
    int client_socket = *(int *) arg;
    char buffer[BUFFER_SIZE];

    while (1) {
        bzero(buffer, BUFFER_SIZE);
        int bytes_received = recv(client_socket, buffer, BUFFER_SIZE, 0);

        if (bytes_received <= 0) {
            break;
        }

        pthread_mutex_lock(&client_mutex);
        for (int i = 0; i < num_clients; i++) {
            if (client_sockets[i] != client_socket) {
                send(client_sockets[i], buffer, bytes_received, 0);
            }
        }
        pthread_mutex_unlock(&client_mutex);
    }

    close(client_socket);
    return NULL;
}

int main() {
    struct sockaddr_in server_addr;
    struct sockaddr_in client_addr;
    socklen_t client_size;

    server_socket = socket(AF_INET, SOCK_STREAM, 0);

    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(8080);
    inet_pton(AF_INET, "127.0.0.1", &server_addr.sin_addr);

    bind(server_socket, (struct sockaddr *) &server_addr, sizeof(server_addr));

    listen(server_socket, 5);

    while (1) {
        client_size = sizeof(client_addr);
        int client_socket = accept(server_socket, (struct sockaddr *) &client_addr, &client_size);

        pthread_mutex_lock(&client_mutex);
        if (num_clients == MAX_CLIENTS) {
            pthread_mutex_unlock(&client_mutex);
            send(client_socket, "Server is full. Please try again later.\n", 46, 0);
            close(client_socket);
            continue;
        }

        client_sockets[num_clients++] = client_socket;
        pthread_mutex_unlock(&client_mutex);

        pthread_create(&threads[num_clients - 1], NULL, handle_client, (void *) &client_socket);
    }

    return 0;
}
