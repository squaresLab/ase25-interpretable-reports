what: The vulnerability is due to the shared variable `client_count` and the array `client_sockets` being accessed by multiple threads without appropriate synchronization mechanisms, leading to a possible race condition.
where: `handle_client` function (starting at line 17) and `main` function (starting at line 39) in file falcon180b-54469.
why: A possible consequence of not resolving this weakness is that the application could experience inconsistent state, leading to application crashes, data corruption, or security breaches.
how: An attacker could manipulate the timing of thread execution to exploit the race condition, causing erratic behavior in the server, including incorrect client socket handling, potential crashes, or unauthorized access.
code-sources: The entry point where user input enters the application is through the `accept` call in the `main` function.
code-sinks: The action performed by the application using user input is in the `handle_client` function through the `recv` and `send` calls.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <pthread.h>

#define MAX_CLIENTS 10
#define BUFFER_SIZE 1024

int server_socket;
int client_sockets[MAX_CLIENTS];
int client_count = 0;
pthread_mutex_t client_mutex = PTHREAD_MUTEX_INITIALIZER;

void *handle_client(void *arg) {
    int client_socket = *(int *) arg;
    char buffer[BUFFER_SIZE];

    while (1) {
        memset(buffer, 0, BUFFER_SIZE);
        int bytes_received = recv(client_socket, buffer, BUFFER_SIZE, 0);

        if (bytes_received <= 0) {
            printf("Client %d disconnected\n", client_socket);
            close(client_socket);

            pthread_mutex_lock(&client_mutex);
            client_count--;
            pthread_mutex_unlock(&client_mutex);

            break;
        }

        printf("Client %d sent: %s\n", client_socket, buffer);

        pthread_mutex_lock(&client_mutex);
        for (int i = 0; i < client_count; i++) {
            if (client_sockets[i] != client_socket) {
                send(client_sockets[i], buffer, strlen(buffer), 0);
            }
        }
        pthread_mutex_unlock(&client_mutex);
    }

    return NULL;
}

int main() {
    int opt = 1;
    setsockopt(server_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    struct sockaddr_in server_addr;
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(3000);
    server_addr.sin_addr.s_addr = INADDR_ANY;

    bind(server_socket, (struct sockaddr *) &server_addr, sizeof(server_addr));
    listen(server_socket, 5);

    while (1) {
        int client_socket = accept(server_socket, NULL, NULL);

        pthread_mutex_lock(&client_mutex);
        if (client_count >= MAX_CLIENTS) {
            pthread_mutex_unlock(&client_mutex);
            printf("Max clients reached\n");
            close(client_socket);
            continue;
        }

        client_sockets[client_count] = client_socket;
        client_count++;
        pthread_mutex_unlock(&client_mutex);

        pthread_t thread_id;
        pthread_create(&thread_id, NULL, handle_client, (void *) &client_socket);
    }

    return 0;
}
