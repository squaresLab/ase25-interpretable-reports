    char buffer[BUFFER_SIZE + 1];  // Allocate additional space for null-terminator

            break;
        }

        buffer[BUFFER_SIZE] = '\0';  // Ensure buffer is null-terminated
        printf("Client %d sent: %s\n", client_socket, buffer);
    }

    close(client_socket);
    return NULL;

void accept_connections() {

            printf("Error creating socket\n");
            exit(1);
