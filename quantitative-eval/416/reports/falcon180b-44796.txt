
vuln-fix: CWE-362: Race Condition in falcon180b-44796 (severity: High)
what: The code suffers from a race condition vulnerability in `handle_client` function where the `client_sockets` array and the `num_clients` variable are accessed by multiple threads without synchronization.
where: The vulnerability is located in the `handle_client` function and the while loop in the `main` function in the falcon180b-44796 file.
why: If not resolved, this race condition could lead to undefined behavior, including possible crashes, corruption of data, and security vulnerabilities such as information disclosure or erroneous program execution.
how: An attacker could potentially exploit this race condition by rapidly connecting and disconnecting clients, causing the server to mishandle the `client_sockets` array and `num_clients` variable, which might lead to overwriting memory or crashing the server application.
code-sources: The source of user input originates from the `recv` function within the `handle_client` function.
code-sinks: The `send` function within the `handle_client` function performs actions based on user input received from the `recv` function.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <errno.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <pthread.h>

#define PORT 8080
#define MAX_CLIENTS 10
#define BUFFER_SIZE 1024

int server_socket;
int client_sockets[MAX_CLIENTS];
int num_clients = 0;
pthread_mutex_t client_mutex = PTHREAD_MUTEX_INITIALIZER;

void *handle_client(void *client_socket) {
    int sock = *(int *) client_socket;
    char buffer[BUFFER_SIZE];
    while (1) {
        memset(buffer, 0, BUFFER_SIZE);
        int bytes_received = recv(sock, buffer, BUFFER_SIZE, 0);
        if (bytes_received <= 0) {
            close(sock);
            pthread_exit(NULL);
        }
        printf("Client %d sent: %s\n", sock, buffer);
        pthread_mutex_lock(&client_mutex);
        for (int i = 0; i < num_clients; i++) {
            if (client_sockets[i] != sock) {
                send(client_sockets[i], buffer, strlen(buffer), 0);
            }
        }
        pthread_mutex_unlock(&client_mutex);
    }
}

int main() {
    int opt = 1;
    setsockopt(server_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in server_addr;
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(PORT);
    server_addr.sin_addr.s_addr = htonl(INADDR_ANY);

    bind(server_socket, (struct sockaddr *) &server_addr, sizeof(server_addr));
    listen(server_socket, MAX_CLIENTS);

    printf("Server started on port %d\n", PORT);

    while (1) {
        struct sockaddr_in client_addr;
        socklen_t addr_len = sizeof(client_addr);
        int client_socket = accept(server_socket, (struct sockaddr *) &client_addr, &addr_len);

        pthread_mutex_lock(&client_mutex);
        if (num_clients >= MAX_CLIENTS) {
            pthread_mutex_unlock(&client_mutex);
            close(client_socket);
            continue;
        }

        client_sockets[num_clients] = client_socket;
        num_clients++;
        pthread_mutex_unlock(&client_mutex);

        pthread_t thread_id;
        pthread_create(&thread_id, NULL, handle_client, (void *) &client_socket);
    }

    return 0;
}
